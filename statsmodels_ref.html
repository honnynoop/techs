statsmodels — 완전 레퍼런스 가이드

statsmodels는 통계 모델 추정·검정·탐색을 위한 파이썬 라이브러리입니다. scikit-learn이 예측(prediction)에 초점을 맞추는 반면, statsmodels는 추론(inference) — 즉 계수의 통계적 유의성, 신뢰구간, p-value, 잔차 진단에 특화되어 있습니다. 빅분기·ADP·ADsP 실기에서 회귀분석, 시계열, 가설검정 코드를 작성할 때 필수입니다.

3 OLS — 최소자승 선형 회귀 Ordinary Least Squares · 가장 기본적이고 중요한 회귀 모델
y = Xβ + ε  |  β̂ = (XᵀX)⁻¹Xᵀy  |  R² = 1 − SSres/SStot  |  AIC = n·ln(SSE/n) + 2k
📐 배열 기반 OLS (sm.OLS)
python
import numpy as np
import pandas as pd
import statsmodels.api as sm

# ── 데이터 준비 ──────────────────────────────────────────
data = sm.datasets.longley.load_pandas().data
y = data['TOTEMP']                        # 종속변수 (1D)
X = data[['GNP', 'UNEMP', 'ARMED', 'POP']] # 독립변수들

# ── 상수항 추가 (필수!) ───────────────────────────────────
X = sm.add_constant(X)  # 첫 열에 1.0 추가 → 절편(β₀) 추정

# ── 모델 정의 및 학습 ─────────────────────────────────────
model  = sm.OLS(y, X)       # 모델 정의
result = model.fit()         # 최소자승 추정

# ── 결과 출력 (매우 상세한 요약) ──────────────────────────
print(result.summary())

# ── 주요 결과값 개별 접근 ─────────────────────────────────
print("계수(β̂):",       result.params)          # Series
print("p-value:",       result.pvalues)         # 각 계수의 유의성
print("R²:",            result.rsquared)        # 결정계수
print("Adj-R²:",        result.rsquared_adj)    # 수정 R²
print("AIC:",           result.aic)             # 아카이케 정보기준
print("BIC:",           result.bic)             # 베이즈 정보기준
print("F-statistic:",   result.fvalue)          # F통계량
print("F p-value:",     result.f_pvalue)        # F검정 p-value
print("MSE(잔차):",     result.mse_resid)       # 잔차 평균제곱오차
print("표준오차:",       result.bse)             # 계수 표준오차
print("95% 신뢰구간:\n", result.conf_int())   # DataFrame

import matplotlib.pyplot as plt
from statsmodels.stats.stattools import durbin_watson
from statsmodels.stats.diagnostic import het_breuschpagan, het_white
from statsmodels.stats.outliers_influence import (
    variance_inflation_factor, OLSInfluence
)

r = result  # OLS fit 결과

# ── 1. 잔차 정규성 검정 ───────────────────────────────────
from scipy.stats import shapiro
stat, p = shapiro(r.resid)
print(f"Shapiro-Wilk: stat={stat:.4f}, p={p:.4f}")
# p > 0.05 → 정규성 만족

# ── 2. 자기상관 검정 (Durbin-Watson) ─────────────────────
dw = durbin_watson(r.resid)
print(f"Durbin-Watson: {dw:.4f}")
# ~2: 자기상관 없음, < 1: 양의 자기상관, > 3: 음의 자기상관

# ── 3. 이분산성 검정 (Breusch-Pagan) ────────────────────
lm, lm_p, f, f_p = het_breuschpagan(r.resid, r.model.exog)
print(f"Breusch-Pagan: LM={lm:.4f}, p={lm_p:.4f}")
# H₀: 등분산  → p < 0.05면 이분산 의심

lm2, lm2_p, _, _ = het_white(r.resid, r.model.exog)
print(f"White test:    LM={lm2:.4f}, p={lm2_p:.4f}")

# ── 4. 다중공선성 — VIF ────────────────────────────────
X_arr = r.model.exog
for i, col in enumerate(r.model.exog_names):
    vif = variance_inflation_factor(X_arr, i)
    print(f"{col:15}: VIF={vif:.2f}")
# VIF > 10 → 심각한 다중공선성

# ── 5. 영향점 분석 (Cook's distance) ─────────────────────
infl = r.get_influence()
cd   = infl.cooks_distance[0]   # Cook's distance 배열
lev  = infl.hat_matrix_diag      # 레버리지
std_resid = infl.resid_studentized_internal  # 스튜던트화 잔차

# 영향점 = Cook's D > 4/n
threshold = 4 / r.nobs
influential = np.where(cd > threshold)[0]

5. anova_lm(model, typ=3) — Type III SS가 표준
⚡ 시계열 단골 출제 (ADP)
1. ADF 검정: p < 0.05 → 정상, p ≥ 0.05 → 비정상 (차분 필요)
2. KPSS는 반대: H₀=정상, p < 0.05 → 비정상
3. ARIMA(p,d,q): ACF→q 결정, PACF→p 결정
4. 계절성: SARIMAX(order, seasonal_order=(P,D,Q,s))
5. res.get_forecast(steps=n) → 예측 + 신뢰구간
⚡ 통계 검정 단골 출제
1. t검정: ws.ttest_mean() / ws.CompareMeans().ttest_ind()
2. 비율: proportions_ztest(count, nobs, value=p0)
3. ANOVA: smf.ols().fit() → anova_lm()
4. 사후검정: pairwise_tukeyhsd(data, groups)
5. 다중검정 보정: multipletests(p_values, method='fdr_bh')

# ═══════════════════════════════════════════════════════

import numpy as np
import pandas as pd
import statsmodels.api as sm
import statsmodels.formula.api as smf
from statsmodels.stats.stattools import durbin_watson, jarque_bera
from statsmodels.stats.diagnostic import het_breuschpagan
from statsmodels.stats.outliers_influence import variance_inflation_factor
from scipy.stats import shapiro

# ── STEP 1: 데이터 준비 ───────────────────────────────
df = pd.read_csv('data.csv')
y = df['target']
X = df[['x1', 'x2', 'x3']]

# ── STEP 2: 다중공선성 사전 확인 (VIF) ───────────────
X_vif = sm.add_constant(X)
for i, col in enumerate(X_vif.columns):
    vif = variance_inflation_factor(X_vif.values, i)
    print(f"VIF({col}): {vif:.2f}")   # >10 → 제거 고려

# ── STEP 3: OLS 모델 ──────────────────────────────────
# 방법 A: sm.OLS (배열 기반)
X_const = sm.add_constant(X)           # ← 반드시!
model   = sm.OLS(y, X_const)
result  = model.fit()

# 방법 B: smf.ols (Formula 기반, 더 편리)
result = smf.ols('target ~ x1 + x2 + x3', data=df).fit()

# ── STEP 4: 결과 확인 ─────────────────────────────────
print(result.summary())

# 핵심 지표 추출
print(f"\nR²:          {result.rsquared:.4f}")
print(f"Adj. R²:     {result.rsquared_adj:.4f}")
print(f"F-stat:      {result.fvalue:.4f}  (p={result.f_pvalue:.4f})")
print(f"AIC:         {result.aic:.4f}")
print(f"BIC:         {result.bic:.4f}")
print("\n계수 & 유의성:")
print(pd.DataFrame({
    'coef':  result.params,
    'std':   result.bse,
    't':     result.tvalues,
    'p':     result.pvalues,
    'sig':   result.pvalues.apply(lambda p:
        '***' if p<0.001 else
        '**'  if p<0.01  else
        '*'   if p<0.05  else
        '.'   if p<0.1   else '')
}).round(4))

# ── STEP 5: 가정 진단 ─────────────────────────────────
print("\n[가정 진단]")

# ① 정규성
sw_stat, sw_p = shapiro(result.resid)
print(f"정규성(Shapiro): p={sw_p:.4f} → {'OK' if sw_p>0.05 else '위반'}")

# ② 자기상관
dw = durbin_watson(result.resid)
print(f"자기상관(D-W):   {dw:.4f} → {'OK' if 1.5)

# ③ 이분산
_, bp_p, _, _ = het_breuschpagan(result.resid, result.model.exog)
print(f"이분산(BP):      p={bp_p:.4f} → {'OK' if bp_p>0.05 else '이분산'}")

# ── STEP 6: 예측 ──────────────────────────────────────
y_hat = result.fittedvalues     # 훈련 데이터 예측값
resid = result.resid             # 잔차

X_new = pd.DataFrame({'x1':[5], 'x2':[10], 'x3':[3]})
pred  = result.get_prediction(sm.add_constant(X_new))
ci    = pred.summary_frame(alpha=0.05)
print("\n예측 결과:")
print(ci[['mean','mean_ci_lower','mean_ci_upper']])
